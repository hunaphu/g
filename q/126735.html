      else {
        val visitQueue = Queue(queue.dequeue())
        val component = Set(visitQueue.head)

          val (x, y) = visitQueue.dequeue()

def find_connected_components(coordinates):
    """Identifies all connected components in a set of coordinates."""
    queue = list(coordinates)
    while queue:
        tile = queue.pop(0)
        component = [tile]
        visit_queue = [tile]
        
        while visit_queue:
            x, y = visit_queue.pop(0)
            for delta_x, delta_y in [(1, 0), (0, 1), (-1, 0), (0, -1)]:
                neighbor = (x + delta_x, y + delta_y)
                if neighbor in queue:
                    visit_queue.append(neighbor)
                    component.append(neighbor)
                    queue.remove(neighbor)
        yield set(component)

def perimeter(component):
    """Calculates the perimeter of a single component."""
    return sum((x + delta_x, y + delta_y) not in component 
               for x, y in component 
               for delta_x, delta_y in [(1, 0), (0, 1), (-1, 0), (0, -1)])

def fill_and_min_perimeter(component):
    """Fills the component to minimize its perimeter by adding one adjacent coordinate at a time."""
    max_coord = max(max(coords) for coords in component)
    grid = {(x, y) for x in range(max_coord + 2) for y in range(max_coord + 2)}
    outside = grid - component
    perimeter_history = [perimeter(component)]
    
    while True:
        fill_candidates = [(x, y) for x, y in outside 
                           if sum((x + delta_x, y + delta_y) in component 
                                  for delta_x, delta_y in [(1, 0), (0, 1), (-1, 0), (0, -1)]) > 1]
        if not fill_candidates:
            break
        new_tile = fill_candidates[0]
        outside.remove(new_tile)
        component.add(new_tile)
        perimeter_history.append(perimeter(component))
    
    return min(perimeter_history)

def process_set_of_coordinates(coordinates):
    """Processes a set of coordinates by finding connected components, and minimizing perimeter for each."""
    components = list(find_connected_components(coordinates))
    return sum(fill_and_min_perimeter(component) for component in components)

How?

We use the recursive function P() to create a list of all possible partitions of the input list. This function is currently sub-optimal, in that it's returning some duplicated partitions -- which does not however alter the final result.

For each partition, we compute the number of fences required to surround all sheep in each group with a rectangle. The sum of these fences gives the score of the partition. We eventually return the lowest score.
