x=>/^((?!(?<!c)ei|cie).)*$/.test(x)  //full program
x=>/                      /.test(x)  //js regex boilerplate
    ^                    $           //within the entire string
     (                .)*            //for every character,
                                     //(if there are any)
      (?!            )               //so long as the next few
                                     //characters are not
                 |                     //either
                  cie                    //"cie"
               ei                        //or an "ei"
         (?<!c)                          //that isnt after "c"
                                     //return true, qed
