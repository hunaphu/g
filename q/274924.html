def f(n):
    arr = [0, 0]
    # the exact number of iterations doesn't matter as long as we generate enough terms
    for i in range(n):
        s = 3 ** i
        for x in range(s, s * 2):
            arr.append(x)
        for x in range(s * 2, s * 3):
            for _ in range(3):
                arr.append(x)
    return arr[n]
