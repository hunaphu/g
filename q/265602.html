  }

  def f(n: Int): String = {
    if (n < 2) return "[]" * n

    var d = Map[Int, Int]()
    var K = List[Int]()
    var num = n

    while (num > 1) {
      breakable({
      for (i <- 2 to num) {
        if (i == 2 || (2 until i).forall(i % _ != 0)) {
          K = i :: K
          if (num % i == 0) {
            num /= i
            d = d.updated(i, d.getOrElse(i, 0) + 1)
            break
          }
        }
      }
      })
    }

    "[" + K.distinct.map(i => f(d.getOrElse(i, 0))).mkString("|") + "]"
  }
