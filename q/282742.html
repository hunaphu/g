Does not work if key or input is over 65 thousand characters.

Readable Version

function g(f, i, k: string): string;
var
  u: word;
  y: string;
  t: word;
begin
  y := '';
  while length(k) < length(i) do
    k := k + k;

  for u := 1 to length(i) do
  begin
    if f = 'd' then
      t := ord(i[u]) div ord(k[u])
    else
      t := ord(i[u]) * ord(k[u]);

    y := y + char(t);
  end;
  Result := y;
end;
