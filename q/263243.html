object Main {

  var c: ArrayBuffer[String] = ArrayBuffer()

  def f(i: String): Int = {
    if (c.isEmpty) {
      c += i
    }
    else {
      c = ArrayBuffer(c.head + i)
    }
    (c.head + c.head(0)).sliding(2).count(_ == "AB")
  }

  def main(args: Array[String]): Unit = {
