// a closure that takes an array of single-digit `Int`s
{
  var
    // the mnemonic list
    // using `split` here saves 6 bytes over an array literal
    t = "NUMBERZERO I BI TRI FOUR FIFTH SEXTET SEVENTH INTEIGHT DIGITNINE".split { $0 < "!" },

    // creates an "XXX:I BI TRI" string from an array of digits
    m = { (d: [Int]) in
      // repeat "X" as many times as there are digits
      // (significantly shorter than `String.init(repeating:count:)`)
      d.map{ _ in "X" }
        // the separating colon
        + ":"
        // the space-separated mnemonics
        + { d.map { t[$0] }.joined}()(" ")
    },
    s: [Int] = $0, // a mutable copy of the closure argument
    x = "PUT " // the string we're building

  // if the number is more than 10 digits...
  while s.count > 10 {
    // ...append the following to the string:
    x +=
      // - the first 10 digits, as mnemonics
      m([] + s[...9])
      // - a newline, "CONTINUED", and a space
      + "\nCONTINUED "

    // remove the first 10 digits
    s = [] + s[10...]
  } // if the number is still more than 10 digits, loop again

  // if the original number was only 1 digit...
  return $0.count < 2
    // get that digit's mnemonic and prepend "PUSH "
    ? "PUSH " + t[s[0]]
    // otherwise, get the string and append the remaining digits as mnemonics
    // (if there were less than 10 digits, the string is still "PUT ")
    : x + m(s)
}
