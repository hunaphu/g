let numBits = parseInt(prompt("Number of Bits: "));
let primes = [];
let notPrime = [0,1];
if (numBits > 1){
  for (let i = 0; i < Math.pow(2,numBits); i++){
    if (!notPrime.includes(i)){
      let binary = i.toString(2).split("");
      if (binary.length < numBits){
        let filler = new Array(numBits - binary.length);
        filler.fill("0");
        binary = filler.concat(binary);
      }
      primes.push(binary);
      for (let j = Math.pow(i,2); j < Math.pow(2,numBits); j += i){
        notPrime.push(j);
      }
    }
  }
}
let rule = "";
for (let i = 0; i < primes.length; i++){
  let numberRule = "(";
  for (let j = 0; j < numBits; j++){
    if (primes[i][j] === "0"){
      numberRule += "(¬d" + (j + 1) + ")";
    }else{
      numberRule += "d" + (j + 1);
    }
    if (j !== numBits - 1){
      numberRule += " & ";
    }
  }
  rule += numberRule + ")";
  if (i !== primes.length - 1){
    rule += " | ";
  }
}

This code was not meant to be efficient, but it has been a bit since the challenge was posted and I figure even this is better than nothing.

The code works by filtering out the primes in the range [0,2^numBits], using the Sieve of Eratosthenes. It converts these primes to binary, and makes a sort of truth table, matching prime binary numbers to 1, and everything else to 0. I then used SOP to fit a boolean expression to this truth table. Here is an example of SOP on a 3-digit prime number table.

Number | Is prime?
==================
000    |0
001    |0
010    |1
011    |1
100    |0
101    |1
110    |0
111    |1

Prime when number = 010 | 011 | 101 | 111

Checking if the number is just one of these cases, 010 for example, can be expanded to (digit 1 = 0) & (digit 2 = 1) & (digit 3 = 0). This can be written more concisely as ((¬d1) & d2 & (¬d3)). Doing this for each case, we get ((¬d1) & d2 & (¬d3)) | ((¬d1) & d2 & d3) | (d1 & (¬d2) & d3) | (d1 & d2 & d3).
