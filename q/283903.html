      [c],                 //     c = leading character

                           //         command is always G9x; so we'll never try
                           //         to use m when it's still undefined)

    c > a ?                //     if c is greater than 'G':

  C = {                    //   start with an object C where ...

)                          // end of map()
&& C                       // return C
