Ungolfed version as regular formula:

=let(
  lcs, lambda(lcs, x, y, let(
    head_, lambda(s, left(s, len(s) - 1)),
    ifs(
      or(x = "", y = ""), "",
      right(x) = right(y), lcs(lcs, head_(x), head_(y)) & right(x),
      true, let(
        a, lcs(lcs, x, head_(y)),
        b, lcs(lcs, head_(x), y),
        if(len(a) < len(b), b, a)
      )
    )
  )),

  lcs(lcs, A1, A2)
)
