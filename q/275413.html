import functools

@functools.lru_cache(maxsize=None)
def tree_count(nodes, max_height):
    if nodes == 1:
        return int(max_height > 0)
    return sum(
        sum(tree_count(i, max_height) * (m := (nodes - i) // d) * tree_count(m, max_height - 1)
            for d in range(1, nodes - i + 1)
            if (nodes - i) % d == 0
            )
        for i in range(1, nodes)
    ) // (nodes - 1)
