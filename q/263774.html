def find_smallest_c(a, b):
    # Find the largest power of 2 that is less than or equal to a
    power_of_two = 1
    while power_of_two * 2 <= a:
        power_of_two *= 2

    # Find the smallest multiple of power_of_two that is greater than or equal to a
    c = power_of_two
    while c < a:
        c += power_of_two

    # Check if c is less than or equal to b
    if c <= b:
        return c
    else:
        return None

# Test cases
