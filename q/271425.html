  let mut a:[i32;12800]=[0;12800];
  let(mut u,mut i)=(String::new(),0);
  for t in x{
    if&"0">t{                            // +, * or /
      i-=1;
      if&"+"==t{                         // + operator
        for m in 1..128{
          if a[128*(i-1)+m]!=a[128*i+m]{
            return format!("*")
          }
        }
        a[128*(i-1)]+=a[128*i]
      }else{
        if&"*"==t{                       // * operator
          a[128*(i-1)]*=a[128*i];
          for m in 1..128{
            a[128*(i-1)+m]+=a[128*i+m]
          }
        }else{                           // / operator
          a[128*(i-1)]/=a[128*i];
          for m in 1..128{
            a[128*(i-1)+m]-=a[128*i+m]
          }
        }
      }
      for m in 1..128{
        a[128*i+m]=0
      }
    }else
      if&"A">t{                          // numbers
        a[128*i]=str::parse::<i32>(t).unwrap();
        i+=1
      }else{                             // units
        a[128*(i-1)+(t.as_bytes()[0]as usize)]+=1
      }
    }
  for t in 0..i{
    u.push_str(&a[128*t].to_string());
    for v in 1..128{
      if a[128*t+v]!=0{
        u.push(v as u8 as char);
        u.push_str(&a[128*t+v].to_string())
      }
    }
    u.push(' ')
  }
  u
}
